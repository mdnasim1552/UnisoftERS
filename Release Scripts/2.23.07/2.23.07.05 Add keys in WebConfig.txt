Add below 3 keys inside AppSettings tag:(Inside <appsettings></appsettings>

<!-- If below set to true or key is missing in webConfig - SE app will Save File content binary blob data into ERS_DocumentStore table -->
    <add key="SaveFileBlobContentToDatabase" value="true" />
    <!-- If below set to false or key is missing in webConfig - SE app will not Save File content in Base64 string format into ERS_DocumentStore table -->
    <add key="SaveBase64FileContentToDatabase" value="false" />

<add key="NIPAPIURL" value="https://dev2.nip.scot.nhs.uk/services/fhirtochi/" />
